const express = require('express'); const bodyParser = require('body-parser'); const path = require('path'); const app = express(); const PORT = process.env.PORT || 3000; // Middleware app.use(bodyParser.json()); app.use(express.static(path.join(__dirname, 'public'))); // Show index.html first app.get('/', (req, res) => { res.sendFile(path.join(__dirname, 'public', 'index.html')); }); // Login API app.post('/login', (req, res) => { const { loginEmail, loginPassword } = req.body; console.log(`Login Email: ${loginEmail}, Password: ${loginPassword}`); res.json({ message: 'Login form submitted! Thank You' }); }); // Signup API app.post('/signup', (req, res) => { const { signupUsername, signupEmail, signupPassword } = req.body; console.log(`Signup Username: ${signupUsername}, Email: ${signupEmail}, Password: ${signupPassword}`); res.json({ message: 'Signup form submitted!' }); }); // Start server app.listen(PORT, () => { console.log(`Server is running on http://localhost:${PORT}`); });
